                use std::hash::{Hash, Hasher};
                use guid_create::GUID;

                impl Hash for GUID {
                    fn hash<H: Hasher>(&self, state: &mut H) {
                        state.write_u32(self.0.data1());
                        state.write_u16(self.0.data2());
                        state.write_u16(self.0.data3());
                        for b in self.0.data4().iter() {
                            state.write_u8(*b)
                        }
                }
